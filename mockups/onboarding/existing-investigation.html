def choose_provider(charge: Charge) -> Provider:
    # Rank providers by per-transaction fee for the given currency
    candidates = [p for p in ALL_PROVIDERS if p.supports(charge)]
    candidates.sort(key=lambda p: p.fee(charge))
    return candidates[0]
